Você pode descrever aqui um fluxo de trabalho típico usando a ferramenta, por exemplo:

  3. O código é gerado, ajustado ou refatorado de forma assistida.

# Exemplo: pequeno validador de senha gerado/refatorado com ajuda da IA

import re

def validar_senha(senha: str) -> list[str] | str:
    erros = []

    if len(senha) < 8:
        erros.append("A senha deve ter pelo menos 8 caracteres.")
    if not re.search(r"[A-Z]", senha):
        erros.append("Inclua ao menos uma letra maiúscula.")
    if not re.search(r"[a-z]", senha):
        erros.append("Inclua ao menos uma letra minúscula.")
    if not re.search(r"\d", senha):
        erros.append("Inclua ao menos um número.")

    return erros or "Senha válida!"


if __name__ == "__main__":
    exemplo = "Senha123"
    resultado = validar_senha(exemplo)
    print(resultado)
